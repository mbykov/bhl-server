package main

import (
	"fmt"
	"log"
	"strings"
	"os"

    ort "github.com/yalue/onnxruntime_go"
    "github.com/sugarme/tokenizer"
    "github.com/sugarme/tokenizer/pretrained"
)

func main() {
	modelDir := "/home/michael/LLM/bhl/Models/vosk-recasepunc-ru-0.22"
	modelPath := "recasepunc.onnx"
	vocabPath := modelDir + "/vocab.txt"
	ortLib := os.Getenv("ORT_HOME") + "/lib/libonnxruntime.so"

	// 1. Working Directory Hack for ONNX .data file
	originalWd, _ := os.Getwd()
	os.Chdir(modelDir)
	defer os.Chdir(originalWd)

	ort.SetSharedLibraryPath(ortLib)
	if !ort.IsInitialized() {
		err := ort.InitializeEnvironment()
		if err != nil {
			log.Fatalf("Init error: %v", err)
		}
	}

	// 2. Session [Input int64, Output float32]
	session, err := ort.NewDynamicSession[int64, float32](modelPath,
		[]string{"input_ids", "attention_mask", "token_type_ids"},
		[]string{"logits"})
	if err != nil {
		log.Fatalf("Session error: %v", err)
	}
	defer session.Destroy()

	// 3. Tokenizer (Load raw vocab.txt)
	wp, err := wordpiece.NewModelFromFile(vocabPath, "[UNK]")
	if err != nil {
		log.Fatalf("Vocab error: %v", err)
	}
	wp.ContinuingSubwordPrefix = "##"
	tk := tokenizer.NewTokenizer(wp)

	inputStr := "привет как дела меня зовут михаил"

	// Create Input
	inputSeq := tokenizer.NewInputSequence(inputStr)
	encodeInput := tokenizer.NewSingleEncodeInput(inputSeq)
	en, err := tk.Encode(encodeInput, true)
	if err != nil {
		log.Fatal(err)
	}

	inputIds := toInt64(en.Ids)
	mask := toInt64(en.AttentionMask)
	types := toInt64(en.TypeIds)
	seqLen := int64(len(inputIds))

	// 4. Input Tensors
	shape := ort.NewShape(1, seqLen)
	in1, _ := ort.NewTensor(shape, inputIds)
	in2, _ := ort.NewTensor(shape, mask)
	in3, _ := ort.NewTensor(shape, types)
	defer in1.Destroy(); defer in2.Destroy(); defer in3.Destroy()

	// 5. Output Tensor
	outputShape := ort.NewShape(1, seqLen, 5)
	outputData := make([]float32, 1*seqLen*5)
	outLogits, _ := ort.NewTensor(outputShape, outputData)
	defer outLogits.Destroy()

	// 6. Run
	err = session.Run([]*ort.Tensor[int64]{in1, in2, in3}, []*ort.Tensor[float32]{outLogits})
	if err != nil {
		log.Fatalf("Inference error: %v", err)
	}

	// 7. Post-process
	logits := outLogits.GetData()
	tokens := en.Tokens
	var result []string

	for i := 0; i < int(seqLen); i++ {
		maxIdx := 0
		maxVal := float32(-1e9)
		for j := 0; j < 5; j++ {
			val := logits[i*5+j]
			if val > maxVal {
				maxVal = val
				maxIdx = j
			}
		}

		word := tokens[i]
		// Skip special tokens
		if word == "[CLS]" || word == "[SEP]" || word == "[UNK]" || word == "[PAD]" {
			continue
		}

		isSubword := strings.HasPrefix(word, "##")
		cleanWord := word
		if isSubword {
			cleanWord = word[2:]
		}

		// Apply Capitalization (Label 4)
		if maxIdx == 4 && !isSubword {
			cleanWord = strings.ToUpper(cleanWord[:1]) + cleanWord[1:]
		}

		if isSubword && len(result) > 0 {
			result[len(result)-1] += cleanWord
		} else {
			result = append(result, cleanWord)
		}

		// Apply Punctuation
		lastIdx := len(result) - 1
		switch maxIdx {
		case 1: result[lastIdx] += ","
		case 2: result[lastIdx] += "."
		case 3: result[lastIdx] += "?"
		}
	}

	fmt.Println("Result:", strings.Join(result, " "))
}

func toInt64(in []int) []int64 {
	out := make([]int64, len(in))
	for i, v := range in { out[i] = int64(v) }
	return out
}
